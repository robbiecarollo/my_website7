Make a plot that shows the top 25 beer consuming countries

{r beer_plot}

beer <- drinks %>% select(country, beer_servings) %>% arrange(desc(beer_servings))

top25_beer <- beer %>% head(25)

ggplot(top25_beer, aes(x= beer_servings, y = reorder(country, beer_servings))) + geom_bar(stat=‘identity’)+ labs(x=“Beer Servings Consumed Per Person”,y = “Country”, title = “Top 25 Beer Consuming Countries”)
